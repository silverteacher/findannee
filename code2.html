// Variables globales

let completedCount = 0;

const totalTasks = document.querySelectorAll('.task-item').length;


// Fonction pour basculer l'état d'une tâche

function toggleTask(taskElement) {

    const checkbox = taskElement.querySelector('input[type="checkbox"]');

    const isCompleted = taskElement.classList.contains('completed');

    

    if (isCompleted) {

        // Décocher la tâche

        taskElement.classList.remove('completed');

        checkbox.checked = false;

        completedCount--;

    } else {

        // Cocher la tâche

        taskElement.classList.add('completed');

        checkbox.checked = true;

        completedCount++;

        

        // Animation de célébration

        taskElement.style.animation = 'none';

        taskElement.offsetHeight; // Force reflow

        taskElement.style.animation = 'fadeIn 0.5s ease';

    }

    

    updateProgress();

    saveProgress();

}


// Fonction pour mettre à jour la barre de progression

function updateProgress() {

    const percentage = Math.round((completedCount / totalTasks) * 100);

    const progressFill = document.getElementById('progressFill');

    const progressText = document.getElementById('progressText');

    const completedTasksSpan = document.getElementById('completedTasks');

    const remainingTasksSpan = document.getElementById('remainingTasks');

    

    progressFill.style.width = percentage + '%';

    completedTasksSpan.textContent = completedCount;

    remainingTasksSpan.textContent = totalTasks - completedCount;

    

    // Messages dynamiques selon la progression

    if (percentage === 0) {

        progressText.textContent = '0% complété - Commencez par cocher vos premières tâches !';

    } else if (percentage < 100) {

        progressText.textContent = percentage + '% complété - Continuez, vous y êtes presque !';

    } else {

        progressText.textContent = '🎉 100% complété - Checklist terminée !';

    }

}


// Fonction pour sauvegarder la progression dans le localStorage

function saveProgress() {

    const completedIndices = [];

    document.querySelectorAll('.task-item').forEach((task, idx) => {

        if (task.classList.contains('completed')) {

            completedIndices.push(idx);

        }

    });

    localStorage.setItem('completedTasks', JSON.stringify(completedIndices));

}


// Fonction pour restaurer la progression depuis le localStorage

function restoreProgress() {

    const completedIndices = JSON.parse(localStorage.getItem('completedTasks') || '[]');

    completedCount = 0;

    document.querySelectorAll('.task-item').forEach((task, idx) => {

        const checkbox = task.querySelector('input[type="checkbox"]');

        if (completedIndices.includes(idx)) {

            task.classList.add('completed');

            checkbox.checked = true;

            completedCount++;

        } else {

            task.classList.remove('completed');

            checkbox.checked = false;

        }

    });

    updateProgress();

}


// Fonction pour réinitialiser toutes les tâches

function resetAllTasks() {

    completedCount = 0;

    document.querySelectorAll('.task-item').forEach(task => {

        task.classList.remove('completed');

        const checkbox = task.querySelector('input[type="checkbox"]');

        checkbox.checked = false;

    });

    updateProgress();

    localStorage.removeItem('completedTasks');

}


// Initialisation au chargement de la page

document.addEventListener('DOMContentLoaded', function() {

    restoreProgress();

});

</script>
